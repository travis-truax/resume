% --- Reusable presentation macros ---
\newcommand{\titlebar}[2]{%
  {\fontsize{\TitleNameSize}{\TitleNameLeading}\selectfont\bfseries\color{ResumeAccent} #1}\par
  \vspace{0.25em}
  \textcolor{ResumeAccent}{\rule{\linewidth}{0.6pt}}\par
  \vspace{0.18em}
  {\small #2}\par
  \vspace{0.38em}
}

\newcommand{\ressection}[1]{%
  \vspace{\SectionTopGap}%
  {\Large\bfseries\color{ResumeAccent} #1}\par
  \vspace{\SectionBottomGap}%
}

\newcommand{\entry}[4]{
  \nopagebreak[3]
  \begin{tabularx}{\textwidth}{@{}Xr@{}}
    {\bfseries\MakeUppercase{#1 \textbar{} #2 \textbar{} #3}} & {\bfseries #4} \\
  \end{tabularx}
  \vspace{\EntryBottomPullup}
  \nopagebreak[3]
}

\newcommand{\coreitem}[2]{%
  {\renewcommand{\arraystretch}{\CoreRowStretch}%
  \begin{tabularx}{\textwidth}{@{}>{\bfseries\raggedright\arraybackslash}p{\CoreLabelWidth}@{\hspace{0.2em}}X@{}}
    {{\CoreTextSize\textcolor{ResumeMuted}{\ResumeBulletGlyph}~\MakeUppercase{#1}}} &
    {{\CoreTextSize\textcolor{ResumeText}{\(\rightarrow\) #2}}} \\
  \end{tabularx}
  }%
  \vspace{\CoreRowBottomPullup}
}

\newenvironment{entryitems}{%
  \vspace{\EntryListTopPullup}
  \begin{itemize}
    \setlength{\topsep}{\ListTopSep}
    \setlength{\partopsep}{\ListPartTopSep}
    \setlength{\parskip}{0pt}
    \setlength{\itemsep}{\ListItemSep}
    \setlength{\parsep}{\ListParseSep}
}{%
  \end{itemize}
}
